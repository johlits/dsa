Course Schedule

Topological Sort BFS Medium

Given course prerequisites, determine whether it is possible to complete all courses. A cycle in the dependency graph makes it impossible.

Approach (Kahn's Algorithm)

Build a directed graph from prerequisites and compute the in-degree of each node. Use a queue to process courses with in-degree 0. For each processed course, reduce the in-degree of its neighbors. If all courses are processed, there is no cycle.

  • Time: O(V + E)
  • Space: O(V + E)
public class Solution {
    public bool CanFinish(int numCourses, int[][] prerequisites) {
        var adj = new List<int>[numCourses];
        for (int i = 0; i < numCourses; i++) adj[i] = new List<int>();
        var inDegree = new int[numCourses];

        foreach (var pre in prerequisites) {
            int course = pre[0], req = pre[1];
            adj[req].Add(course);
            inDegree[course]++;
        }

        var queue = new Queue<int>();
        for (int i = 0; i < numCourses; i++)
            if (inDegree[i] == 0) queue.Enqueue(i);

        int processed = 0;
        while (queue.Count > 0) {
            int curr = queue.Dequeue();
            processed++;
            foreach (int next in adj[curr]) {
                inDegree[next]--;
                if (inDegree[next] == 0) queue.Enqueue(next);
            }
        }

        return processed == numCourses;
    }
}
