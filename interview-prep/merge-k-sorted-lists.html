Merge k Sorted Lists

Heap Linked List Hard

Merge k already-sorted linked lists into one sorted linked list efficiently.

Approach (Min-Heap)

Insert the first node of each list into a min-heap. Repeatedly extract the smallest node and append it to the result, then insert the next node from that list into the heap. This avoids the O(k · n) cost of comparing all heads each time.

  • Time: O(N log k) where N is total nodes, k is number of lists.
  • Space: O(k) for the heap.
public class ListNode {
    public int val;
    public ListNode next;
    public ListNode(int val = 0, ListNode next = null) {
        this.val = val; this.next = next;
    }
}

public class Solution {
    public ListNode MergeKLists(ListNode[] lists) {
        var heap = new PriorityQueue<ListNode, int>();
        foreach (var list in lists) {
            if (list != null) heap.Enqueue(list, list.val);
        }

        var dummy = new ListNode();
        var tail = dummy;

        while (heap.Count > 0) {
            var node = heap.Dequeue();
            tail.next = node;
            tail = node;
            if (node.next != null) {
                heap.Enqueue(node.next, node.next.val);
            }
        }

        return dummy.next;
    }
}
