Clone Graph

Graph BFS Medium

Given a node in a connected graph, create a deep copy of the entire graph while preserving all connections.

Approach

Use BFS starting from the given node. Maintain a dictionary that maps original nodes to their clones. For each original node, create a clone if it doesn't exist, then clone its neighbors. The graph is undirected and may contain cycles, so the visited map prevents infinite loops.

  • Time: O(V + E) — every node and edge is visited once.
  • Space: O(V) for the clone map and queue.
public class Node {
    public int val;
    public IList<Node> neighbors;
    public Node(int val = 0, IList<Node> neighbors = null) {
        this.val = val;
        this.neighbors = neighbors ?? new List<Node>();
    }
}

public class Solution {
    public Node CloneGraph(Node node) {
        if (node == null) return null;

        var visited = new Dictionary<Node, Node>();
        var queue = new Queue<Node>();
        queue.Enqueue(node);
        visited[node] = new Node(node.val);

        while (queue.Count > 0) {
            Node curr = queue.Dequeue();
            foreach (Node neighbor in curr.neighbors) {
                if (!visited.ContainsKey(neighbor)) {
                    visited[neighbor] = new Node(neighbor.val);
                    queue.Enqueue(neighbor);
                }
                visited[curr].neighbors.Add(visited[neighbor]);
            }
        }

        return visited[node];
    }
}
