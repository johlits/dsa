Coin Change

DP Medium

Given coin denominations and a target amount, determine the minimum number of coins needed to make that amount, or report that it's impossible.

Approach (Bottom-up DP)

Let dp[i] be the minimum coins needed for amount i. Initialize with infinity (or amount + 1). For each coin, update all reachable amounts. If dp[amount] remains infinity, return -1.

  • Time: O(A · C) where A = amount, C = number of coins.
  • Space: O(A)
public class Solution {
    public int CoinChange(int[] coins, int amount) {
        int max = amount + 1;
        var dp = new int[max];
        Array.Fill(dp, max);
        dp[0] = 0;

        for (int i = 1; i <= amount; i++) {
            foreach (int coin in coins) {
                if (coin <= i) {
                    dp[i] = Math.Min(dp[i], dp[i - coin] + 1);
                }
            }
        }

        return dp[amount] > amount ? -1 : dp[amount];
    }
}
