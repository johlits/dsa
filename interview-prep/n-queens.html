N-Queens

Backtracking Hard

Place n queens on an n × n chessboard so that no two queens attack each other. Return all distinct solutions.

Approach

Use backtracking: place one queen per row. For each row, try every column and check if the position conflicts with already placed queens (same column or diagonal). If valid, recurse to the next row. When all rows are filled, record the board configuration.

  • Time: O(N!) — roughly N choices, then N-2, etc.
  • Space: O(N) for the recursion stack and board state.
public class Solution {
    public IList<IList<string>> SolveNQueens(int n) {
        var result = new List<IList<string>>();
        var board = new char[n][];
        for (int i = 0; i < n; i++) {
            board[i] = new char[n];
            Array.Fill(board[i], '.');
        }
        Backtrack(0);
        return result;

        void Backtrack(int row) {
            if (row == n) {
                var copy = new List<string>();
                foreach (var r in board) copy.Add(new string(r));
                result.Add(copy);
                return;
            }
            for (int col = 0; col < n; col++) {
                if (IsValid(row, col)) {
                    board[row][col] = 'Q';
                    Backtrack(row + 1);
                    board[row][col] = '.';
                }
            }
        }

        bool IsValid(int row, int col) {
            for (int i = 0; i < row; i++) {
                if (board[i][col] == 'Q') return false;
            }
            for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
                if (board[i][j] == 'Q') return false;
            }
            for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
                if (board[i][j] == 'Q') return false;
            }
            return true;
        }
    }
}
