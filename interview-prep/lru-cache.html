LRU Cache

Hash Map Linked List Medium

Design a cache with a fixed capacity that supports O(1) lookups and updates while evicting the least recently used item when full.

Approach

Use a doubly linked list to store items in usage order (most recent at head, least recent at tail) and a dictionary for O(1) key lookups. On get/put, move the accessed node to the head. If capacity is exceeded, remove the tail node.

  • Get/Put Time: O(1)
  • Space: O(capacity)
public class LRUCache {
    private class Node {
        public int Key, Value;
        public Node Prev, Next;
        public Node(int k, int v) { Key = k; Value = v; }
    }

    private readonly int capacity;
    private readonly Dictionary<int, Node> map;
    private Node head, tail;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        map = new Dictionary<int, Node>(capacity);
        head = new Node(0, 0);
        tail = new Node(0, 0);
        head.Next = tail;
        tail.Prev = head;
    }

    public int Get(int key) {
        if (!map.TryGetValue(key, out Node node)) return -1;
        MoveToHead(node);
        return node.Value;
    }

    public void Put(int key, int value) {
        if (map.TryGetValue(key, out Node node)) {
            node.Value = value;
            MoveToHead(node);
            return;
        }
        if (map.Count >= capacity) {
            map.Remove(tail.Prev.Key);
            Remove(tail.Prev);
        }
        var newNode = new Node(key, value);
        map[key] = newNode;
        AddToHead(newNode);
    }

    private void AddToHead(Node node) {
        node.Prev = head;
        node.Next = head.Next;
        head.Next.Prev = node;
        head.Next = node;
    }

    private void Remove(Node node) {
        node.Prev.Next = node.Next;
        node.Next.Prev = node.Prev;
    }

    private void MoveToHead(Node node) {
        Remove(node);
        AddToHead(node);
    }
}
