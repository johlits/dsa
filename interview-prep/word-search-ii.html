Word Search II

Trie Backtracking Hard

Given a grid of letters and a list of words, find all words that can be formed by traversing adjacent cells without reusing a cell in the same word.

Approach

Build a Trie from the word list. For each cell in the board, run DFS using the Trie to prune paths that cannot match any word. When a complete word is found, add it to the result and remove it from the Trie to avoid duplicates.

  • Time: O(M · N · 4L) worst case, but Trie pruning makes it much faster in practice.
  • Space: O(K · L) for the Trie, where K is number of words and L is average length.
public class Solution {
    private class TrieNode {
        public TrieNode[] Children = new TrieNode[26];
        public string Word;
    }

    private TrieNode root;
    private char[][] board;
    private int m, n;
    private IList<string> result = new List<string>();

    public IList<string> FindWords(char[][] board, string[] words) {
        this.board = board;
        m = board.Length;
        n = board[0].Length;
        root = new TrieNode();
        foreach (var word in words) {
            var node = root;
            foreach (var ch in word) {
                int idx = ch - 'a';
                if (node.Children[idx] == null) node.Children[idx] = new TrieNode();
                node = node.Children[idx];
            }
            node.Word = word;
        }

        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                Dfs(i, j, root);

        return result;
    }

    private void Dfs(int i, int j, TrieNode node) {
        if (i < 0 || i >= m || j < 0 || j >= n) return;
        char ch = board[i][j];
        if (ch == '#') return;
        int idx = ch - 'a';
        if (node.Children[idx] == null) return;

        node = node.Children[idx];
        if (node.Word != null) {
            result.Add(node.Word);
            node.Word = null; // avoid duplicates
        }

        board[i][j] = '#';
        Dfs(i + 1, j, node);
        Dfs(i - 1, j, node);
        Dfs(i, j + 1, node);
        Dfs(i, j - 1, node);
        board[i][j] = ch;
    }
}
