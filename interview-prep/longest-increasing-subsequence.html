Longest Increasing Subsequence

Binary Search DP Medium

Given an array of numbers, find the length of the longest subsequence whose values are strictly increasing.

Approach (Patience Sorting / Binary Search)

Maintain a list where each element is the smallest tail of all increasing subsequences of a given length. For each number, binary search its position in this list. If it extends the list, append it; otherwise, replace the first element that is ≥ it. The length of the list is the answer.

  • Time: O(N log N)
  • Space: O(N)
public class Solution {
    public int LengthOfLIS(int[] nums) {
        var tails = new List<int>();
        foreach (int num in nums) {
            int i = BinarySearch(tails, num);
            if (i == tails.Count) tails.Add(num);
            else tails[i] = num;
        }
        return tails.Count;
    }

    private int BinarySearch(List<int> list, int target) {
        int lo = 0, hi = list.Count;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (list[mid] < target) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
}
