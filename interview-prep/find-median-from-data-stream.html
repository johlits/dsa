Find Median from Data Stream

Heap Hard

Continuously receive numbers and efficiently return the median after each insertion.

Approach (Two Heaps)

Maintain a max-heap for the lower half and a min-heap for the upper half. Ensure the heaps are balanced (sizes differ by at most 1). The median is the top of the larger heap, or the average of both tops if they are equal size.

  • AddNum Time: O(log N)
  • FindMedian Time: O(1)
  • Space: O(N)
public class MedianFinder {
    // max-heap for the lower half (stored as negative values)
    private PriorityQueue<int, int> low;
    // min-heap for the upper half
    private PriorityQueue<int, int> high;

    public MedianFinder() {
        low = new PriorityQueue<int, int>(Comparer<int>.Default);
        high = new PriorityQueue<int, int>();
    }

    public void AddNum(int num) {
        if (low.Count == 0 || num <= low.Peek()) {
            low.Enqueue(num, -num);
        } else {
            high.Enqueue(num, num);
        }

        // Rebalance
        if (low.Count > high.Count + 1) {
            int val = low.Dequeue();
            high.Enqueue(val, val);
        } else if (high.Count > low.Count) {
            int val = high.Dequeue();
            low.Enqueue(val, -val);
        }
    }

    public double FindMedian() {
        if (low.Count > high.Count) return low.Peek();
        return (low.Peek() + high.Peek()) / 2.0;
    }
}
